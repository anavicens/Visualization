import matplotlib.pyplot as plt
import numpy as np
import plotly.express as px
import pandas as pd
import plotly.graph_objects as go

df_ = pd.read_excel("poblacion.xlsx")

data = {'Age': range(0, 110, 10),
        'women': df_[df_['Ao']==2022]['Mujeres'].to_list(),
        'men': df_[df_['Ao']==2022]['Hombres'].to_list()}
df = pd.DataFrame(data)

fig = go.Figure()

fig.add_trace(go.Bar(x=df['women'], y=df['Age'], orientation='h', name='Women', marker_color='skyblue'))
fig.add_trace(go.Bar(x=-df['men'], y=df['Age'], orientation='h', name='Men', marker_color='orange'))

                  xaxis_title='Population', yaxis_title='Age',
                  title='Population Pyramid')

fig.update_xaxes(tickvals=-df['men'], ticktext=df['men'])

fig.show()

import matplotlib.pyplot as plt
import numpy as np
import pandas as pd
import matplotlib.colors as mcolors

df = pd.read_excel("temperaturas.xlsx")


temperaturas = df['Temperatura Media (1)']
years = df['Ao'].unique()


theta = np.linspace(0, 2 * np.pi, 12, endpoint=False)


fig = plt.figure(figsize=(8,8))
ax = plt.subplot(111, projection = 'polar')

colors = plt.cm.tab10.colors 

for i, year in enumerate(years):
    temperatures_year = temperaturas[df['Ao'] == year]
    normalized_temperatures = (temperatures_year - temperatures_year.min()) / (temperatures_year.max() - temperatures_year.min())
    color = plt.cm.tab10(i % len(years))
    theta_repeat = np.tile(theta, len(temperatures_year))
    theta_i = theta_repeat.reshape(-1, 12).tolist()
    ax.plot(theta_i[i], normalized_temperatures, label=str(year), color=color)

meses_ticks = np.linspace(0, 2 * np.pi, 12, endpoint=False)
ax.set_xticks(meses_ticks)
ax.set_xticklabels(['Ene', 'Feb', 'Mar', 'Abr', 'May', 'Jun', 'Jul', 'Ago', 'Sep', 'Oct', 'Nov', 'Dic'])


ax.set_rticks([])  
ax.set_yticklabels([])  
ax.set_title('Temperatura Media Mensual a lo largo de 10 Aos', va='bottom')
ax.legend()


plt.show()

import matplotlib.pyplot as plt
from matplotlib.patches import Ellipse
import numpy as np


columnas_azules_1 = 5
columnas_azules_2 = 5
columnas_azules_3 = 3
columnas_rojos_3 = 2
columnas_rojos_4 = 5
columnas_rojos_5 = 5
tamano_cuadradito = 0.25
espaciado_vertical = 0.20 


fig, ax = plt.subplots()


for i in range(columnas_azules_1):
    circulo_azul_1 = Ellipse((i * tamano_cuadradito + tamano_cuadradito / 2, - espaciado_vertical), tamano_cuadradito * 0.5, tamano_cuadradito * 0.7, color='blue', alpha=0.8)
    ax.add_patch(circulo_azul_1)

for i in range(columnas_azules_2):
    circulo_azul_2 = Ellipse((i * tamano_cuadradito + tamano_cuadradito / 2, -2 *espaciado_vertical), tamano_cuadradito * 0.5, tamano_cuadradito * 0.7, color='blue', alpha=0.8)
    ax.add_patch(circulo_azul_2)

for i in range(columnas_azules_3):
    circulo_azul_3 = Ellipse((i * tamano_cuadradito + tamano_cuadradito / 2, -3 * espaciado_vertical), tamano_cuadradito * 0.5, tamano_cuadradito * 0.7, color='blue', alpha=0.8)
    ax.add_patch(circulo_azul_3)

for i in range(columnas_rojos_3):
    circulo_rojo_3 = Ellipse(((columnas_azules_3 + i) * tamano_cuadradito + tamano_cuadradito / 2, -3 * espaciado_vertical), tamano_cuadradito * 0.5, tamano_cuadradito * 0.7, color='red', alpha=0.8)
    ax.add_patch(circulo_rojo_3)

for i in range(columnas_rojos_4):
    circulo_rojo_4 = Ellipse((i * tamano_cuadradito + tamano_cuadradito / 2, -4* espaciado_vertical), tamano_cuadradito * 0.5, tamano_cuadradito * 0.7, color='red', alpha=0.8)
    ax.add_patch(circulo_rojo_4)

for i in range(columnas_rojos_5):
    circulo_rojo_5 = Ellipse((i * tamano_cuadradito + tamano_cuadradito / 2, -5 * espaciado_vertical), tamano_cuadradito * 0.5, tamano_cuadradito * 0.7, color='red', alpha=0.8)
    ax.add_patch(circulo_rojo_5)


ax.set_axis_off() 
ax.set_xlim(0, max(columnas_azules_1, columnas_azules_2, columnas_azules_3 + columnas_rojos_3, columnas_rojos_4, columnas_rojos_5) * tamano_cuadradito)
ax.set_ylim(-6 * espaciado_vertical, 4 *espaciado_vertical)
ax.legend(handles=[Ellipse((0, 0), 1, 1, color='blue', alpha=0.8), Ellipse((0, 0), 1, 1, color='red', alpha=0.8)],
          labels=['Hombres', 'Mujeres'],
          loc='upper center', bbox_to_anchor=(0.5, -0.15), fancybox=True, shadow=True, ncol=2)
plt.title('Distribucin de Hombres y Mujeres en 2022 con 20 aos en Espaa')

plt.show()

import matplotlib.pyplot as plt
import numpy as np

# Datos de ejemplo
edades = np.arange(0, 101, 10)  # Edades en mltiplos de 10 desde 0 hasta 100
datos_hombres = [5, 10, 15, 20, 25, 30, 35, 40, 35, 30, 25, 20, 15, 10, 5]
datos_mujeres = [15, 10, 5, 0, 5, 10, 15, 20, 25, 30, 35, 40, 35, 30, 25]


fig, ax = plt.subplots()


for edad, hombres, mujeres in zip(edades, datos_hombres, datos_mujeres):
    y_hombres = np.arange(hombres)
    y_mujeres = np.arange(mujeres)
    ax.scatter(np.full_like(y_hombres, edad), y_hombres, color='red', alpha=0.8)
    ax.scatter(np.full_like(y_mujeres, edad), y_mujeres + hombres, color='blue', alpha=0.8)



ax.set_xlabel('Edades')
ax.set_ylabel('Cantidad')
ax.legend(['Hombres', 'Mujeres'])
ax.grid(True)


plt.show()
